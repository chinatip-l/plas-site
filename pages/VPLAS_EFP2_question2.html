module Comp_DFF(
    output reg Q,
    input CLK,Reset
);
    always @ (negedge CLK, posedge Reset)
    begin
        if(Reset)
             <=1'b0; //1'b0 means a 1 bit binary number 0;
        else
             <= ~Q;
        end
endmodule
